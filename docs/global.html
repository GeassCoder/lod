debounce(fn, waitopt) → {function}

Creates a debounced version of fn. When the debounced version is called multiple times fn will not be called until the last attempt has elapsed more than the wait period.
Example
let count = 0;

function fn () {
    ++count;
}

let debouncedFn = debounce(fn, 200);

debouncedFn();
debouncedFn();
debouncedFn();
console.log(count); // logs 0
// although debouncedFn is called multiple times, fn is not called yet because it's not been 200ms yet

setTimeout(() => {
    debouncedFn();
    debouncedFn();
    console.log(count); // logs 0
// fn is still not called yet because it's not been 200ms yet
}, 100);

setTimeout(() => {
    console.log(count); // logs 1
// fn is called as it has passed 200ms since the last attempt (which occured at the 100th ms)
}, 301);
Parameters:
Name Type Attributes Default Description
fn function The function to be debounced.
wait Number <optional>
 100 The time to wait.
Returns:
The debounced version of fn.
Type
function

groupBy(arr, iterateeopt) → {Object}

Creates an object composed of keys generated from the results of running each element of arr thru iteratee. The order of grouped values is determined by the order they occur in arr. The corresponding value of each key is an array of elements responsible for generating that key. iteratee is invoked with one argument: (value).
Example
let result = groupBy(
    [3, 21, 34, 5, 2, 4, 3, 1, 2, 1],
    val => val % 2 ? "odd" : "even"
);

// This logs {
//     odd: [3, 21, 5, 3, 1, 1],
//     even: [34, 2, 4, 2]
// }
console.log(result);
Parameters:
Name Type Attributes Default Description
arr Array The array to iterate over.
iteratee function <optional>
 identity The iteratee to generate keys.
Returns:
The composed object with elements grouped under its keys.
Type
Object

pick(obj, …paths) → {Object}

Creates an object composed of the picked obj properties. If the value of a given path is null or undefined or the last part of the path is is not the corresponding object's own property, that path will be ignored.
Example
let obj = {
    foo: [{
        id: "#1",
        value: 1
    }, {
        id: "#2",
        value: 2
    }],
    x: 55,
    y: {
        id: "y",
        value: "yy"
    }
};

let result = pick(obj, "foo.0.value", "foo.notExist", "y");

// This logs {
//     foo: [{
//         value: 1
//     }],
//     y: {
//         id: "y",
//         value: "yy"
//     }
// }
console.log(result);
Parameters:
Name Type Attributes Description
obj Object The source object.
paths String <repeatable>
 The paths from where values should be picked.
Returns:
The new object with picked values at given ...paths.
Type
Object

range(startopt, end, stepopt) → {Array}

Generate an array of numbers with in the specified range.

If only 1 argument is passed in, it will be used as end, start will be defaulted to 0 and step will be defaulted to 1 or -1 whichever makes sense.

If only 2 arguments are passed in, they will be used as start and end, and step will be defaulted to 1 or -1 whichever makes sense.
Example
range(3); // [0, 1, 2, 3]
range(-3); // [0, -1, -2, -3]
range(1, 3); // [1, 2, 3]
range(1, -3); // [1, 0, -1, -2, -3]
range(1, 4, 2); // [1, 3]
range(1, -4, 2); // [1, -1, -3]
Parameters:
Name Type Attributes Default Description
start Number <optional>
 0 The start point of the range.
end Number The end point of the range.
step Number <optional>
 1 | -1 The step of the range.
Returns:
An array of numbers within the specified range.
Type
Array

sortBy(arr, iteratee, isDescendingopt) → {Array}

Sort given array arr based on {@iteratee} and return a new array (shallow copy) so that the original array arr is untouched.
Example
let arr = [1, 3, 2];
sortBy(arr); // [1, 2, 3]

arr = [{value: 1}, {value: 3}, {value: 2}];
sortBy(arr, element => element.value); // [{value: 1}, {value: 2}, {value: 3}]

arr = [{value: 1}, {value: 3}, {value: 2}];
sortBy(arr, "value", true); // [{value: 3}, {value: 2}, {value: 1}]

arr = [{
    inner: {
        value: 3
    }
}, {
    inner: {
        value: 1
    }
}];
sortBy(arr, "inner.value"); // [{value: 1}, {value: 3}]
Parameters:
Name Type Attributes Default Description
arr Array The array to be sorted.
iteratee function | String The iteratee for sorting if it's a function; the property path in each element if it's a string.
isDescending Boolean <optional>
 false Whether or not to sort in descending order.
Returns:
A new array with elements sorted based on {@iteratee}.
Type
Array

uniqBy(arr, iterateeopt) → {Array}

Iterate over the array arr, de-duplicate it based on iteratee and return a new array (shallow copy).
Example
uniqBy([1, 3, 3, 2, 1, 1, 4, 1, 2, 3]); // [1, 3, 2, 4]

let arr = [{
    value: 1
}, {
    value: 2
}, {
    value: 1
}];

// This logs [{
//     value: 1
// }, {
//     value: 2
// }]
console.log(uniqBy(arr, element => element.value));
